import { theme } from "../tailwind.config.js";
import { Meta, ColorPalette, ColorItem } from "@storybook/addon-docs";
import defaultConfig from "tailwindcss/defaultConfig";
import resolveConfig from "tailwindcss/resolveConfig";

export const colors = resolveConfig([theme, defaultConfig])[0].colors;

<Meta title="Framework/Colors" />

# Colors

<ColorPalette>
  {Object.entries(colors)
    .filter(([key, value]) => !["current", "transparent"].includes(key))
    .map(([key, value]) => {
      return (
        <ColorItem
          title={key}
          subtitle={typeof value === "string" ? value : ""}
          colors={typeof value === "string" ? { [key]: value } : value}
        />
      );
    })}
</ColorPalette>
